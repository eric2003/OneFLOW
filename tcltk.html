modified cgnsview.tcl

proc file_stats {} {
  global ProgData
  set ProgData(file,size) "[file size $ProgData(file,name)] bytes"
  if {[catch {CGIOpen $ProgData(file)} msg]} {
    return $msg
  }
  set ProgData(file,type) [CGNSfile $ProgData(file)]
  set ProgData(file,vers) [CGIOversion]
